    // test getters for beta audio engine
    public static MusicTest getMusicBeta (String key){
        if (key != "music" && key != "null" && key != "shop_music" && key != "game_over"
                && game_.gameState != STATE.GameBeta) {
            if (currentMusic != null) streamPlayer.stop();
            currentMusic = key;
            game_.gameState = STATE.Load;
            game_.stringsforloading = "loading music: " + key;
            game_.loadstate -= 25;
            load(key);
            game_.loadstate += 25;
        }
        return musicTestMap.get(key);
    }
